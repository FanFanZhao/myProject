JS代码

function checkLength(which) {
  var maxChars = 120;
  if (which.value.length > maxChars)
  which.value = which.value.substring(0,maxChars);
  var curr = maxChars - which.value.length;
  document.getElementById("currentLength").innerHTML = curr.toString()}
